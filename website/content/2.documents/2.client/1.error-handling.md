The mion client provides powerful typed error handling through two main classes: `TypedPromise` for route errors and `TypedEvent` for hook errors. This enables precise error handling with full type safety.

## Success-Only Return Types

One of the key features of the mion client is that **routes always resolve to the success type only**, even when the server-side handler returns a union that includes `RpcError` types.

:::code-group
```ts [Client]
import {initClient} from '@mionkit/client';
import type {MyApi} from './server.routes';

const {routes} = initClient<MyApi>({baseURL: 'http://localhost:3000'});

// The resolved type is User, not User | RpcError
const user = await routes.users.getById('123').call();
console.log(user.name); // TypeScript knows user is User, not RpcError!
```

```ts [server.routes.ts]
import {RpcError} from '@mionkit/core';
import {Routes, route, initMionRouter} from '@mionkit/router';

type User = {id: string; name: string};

const routes = {
    users: {
        // Server returns User | RpcError, but client only sees User
        getById: route((ctx, id: string): User | RpcError<'user-not-found', {userId: string}> => {
            const user = findUser(id);
            if (!user) {
                return new RpcError({type: 'user-not-found', errorData: {userId: id}});
            }
            return user;
        }),
    },
} satisfies Routes;

export const myApi = initMionRouter(routes);
export type MyApi = typeof myApi;
```
:::

This design enables **cleaner sequential code** where the happy path flows naturally, and errors are handled separately via dedicated methods.

## Sequential Async/Await Pattern

Because routes always resolve to success types, you can write multiple sequential calls with clean, readable code. Errors are handled separately, keeping the happy path uncluttered:

:::code-group
```ts [Client]
import {initClient} from '@mionkit/client';
import type {MyApi} from './server.routes';

const {routes} = initClient<MyApi>({baseURL: 'http://localhost:3000'});

async function processOrder(userId: string, productId: string) {
    // Happy path - clean sequential code
    const user = await routes.users.getById(userId).call()
        .catchError('user-not-found', (e) => {
            showError(`User ${e.errorData?.userId} not found`);
        });

    const product = await routes.products.getById(productId).call()
        .catchError('product-not-found', (e) => {
            showError(`Product not available`);
        })
        .catchError('out-of-stock', (e) => {
            showWarning(`Only ${e.errorData?.available} items left`);
        });

    const order = await routes.orders.create({userId, productId, quantity: 1}).call()
        .catchError('insufficient-funds', (e) => {
            showError(`Payment failed: need $${e.errorData?.required}`);
        })
        .catchError('cart-empty', () => {
            showWarning('Your cart is empty');
        })
        .catchUnknown((e) => {
            logError(e);
            showError('Something went wrong');
        });

    // All success - continue with order confirmation
    showSuccess(`Order ${order.id} created for ${user.name}!`);
    return order;
}
```

```ts [server.routes.ts]
import {RpcError} from '@mionkit/core';
import {Routes, route, initMionRouter} from '@mionkit/router';

type User = {id: string; name: string};
type Product = {id: string; name: string; stock: number};
type Order = {id: string; userId: string; productId: string};

const routes = {
    users: {
        getById: route((ctx, id: string): User | RpcError<'user-not-found', {userId: string}> => {
            const user = findUser(id);
            if (!user) return new RpcError({type: 'user-not-found', errorData: {userId: id}});
            return user;
        }),
    },
    products: {
        getById: route((ctx, id: string): Product | RpcError<'product-not-found'> | RpcError<'out-of-stock', {available: number}> => {
            const product = findProduct(id);
            if (!product) return new RpcError({type: 'product-not-found'});
            if (product.stock === 0) return new RpcError({type: 'out-of-stock', errorData: {available: 0}});
            return product;
        }),
    },
    orders: {
        create: route((ctx, data: {userId: string; productId: string; quantity: number}): Order | RpcError<'insufficient-funds', {required: number}> | RpcError<'cart-empty'> => {
            // Order creation logic...
            return {id: 'ORDER-123', userId: data.userId, productId: data.productId};
        }),
    },
} satisfies Routes;

export const myApi = initMionRouter(routes);
export type MyApi = typeof myApi;
```
:::

::note
When an error occurs and is handled by `catchError` or `catchUnknown`, the promise resolves to `undefined`. You may want to add early returns or null checks depending on your use case.
::

## Route Error Handling with TypedPromise

When you call a route using `.call()`, it returns a `TypedPromise` that provides chainable error handling methods.

### Basic Usage

:::code-group
```ts [Client]
import {initClient} from '@mionkit/client';
import type {MyApi} from './server.routes';

const {routes} = initClient<MyApi>({baseURL: 'http://localhost:3000'});

// Call a route with typed error handling
routes.users
    .getById('123')
    .call()
    .then((user) => {
        // Success - user is fully typed, never an RpcError
        console.log(user.name);
    })
    .catchError('user-not-found', (error) => {
        // Handle specific error - error.errorData is fully typed!
        console.log('User not found:', error.errorData?.userId);
    })
    .catchUnknown((error) => {
        // Handle any error not caught by catchError
        console.log('Unexpected error:', error.publicMessage);
    })
    .finally(() => {
        // Always runs - cleanup logic
        setLoading(false);
    });
```

```ts [server.routes.ts]
import {RpcError} from '@mionkit/core';
import {Routes, route, initMionRouter} from '@mionkit/router';

type User = {id: string; name: string};

const routes = {
    users: {
        getById: route((ctx, id: string): User | RpcError<'user-not-found', {userId: string}> => {
            const user = findUser(id);
            if (!user) return new RpcError({type: 'user-not-found', errorData: {userId: id}});
            return user;
        }),
    },
} satisfies Routes;

export const myApi = initMionRouter(routes);
export type MyApi = typeof myApi;
```
:::

### TypedPromise Methods

#### `.catchError(errorType, handler)`

Handles a specific error type. The error parameter is fully typed based on your route's return type.

:::code-group
```ts [Client]
routes.orders
    .getById('ORDER-404')
    .call()
    .catchError('order-not-found', (error) => {
        // error.errorData is strongly typed as OrderNotFoundData!
        console.log('Order not found. Requested ID:', error.errorData?.requestedId);
    });
```

```ts [server.routes.ts]
type Order = {id: string; total: number};
type OrderNotFoundData = {requestedId: string};

const routes = {
    orders: {
        getById: route((ctx, id: string): Order | RpcError<'order-not-found', OrderNotFoundData> => {
            const order = findOrder(id);
            if (!order) return new RpcError({type: 'order-not-found', errorData: {requestedId: id}});
            return order;
        }),
    },
} satisfies Routes;
```
:::

#### `.catchUnknown(handler)`

Handles any error not caught by a specific `catchError` handler. Hook errors handled by `onError()` are suppressed and won't reach `catchUnknown`.

:::code-group
```ts [Client]
routes.users
    .sayHello(user)
    .call()
    .catchUnknown((error) => {
        // Catches any unhandled errors
        console.log('Error:', error.publicMessage);
    });
```

```ts [server.routes.ts]
type User = {id: string; name: string};

const routes = {
    users: {
        // This route doesn't define specific error types
        sayHello: route((ctx, user: User): string => {
            return `Hello ${user.name}!`;
        }),
    },
} satisfies Routes;
```
:::

#### `.catch(handler)`

Receives a record of all unhandled errors keyed by method/hook ID. Useful for centralized error logging.

:::code-group
```ts [Client]
routes.users
    .getById('123')
    .call()
    .catchError('user-not-found', (e) => console.log(e))
    .catch((errors) => {
        // errors: { 'users.getById'?: RpcError, 'auth'?: RpcError, ... }
        for (const [methodId, error] of Object.entries(errors)) {
            console.log(`Unhandled error from ${methodId}:`, error);
        }
    });
```

```ts [server.routes.ts]
type User = {id: string; name: string};

const routes = {
    users: {
        getById: route((ctx, id: string): User | RpcError<'user-not-found'> => {
            const user = findUser(id);
            if (!user) return new RpcError({type: 'user-not-found'});
            return user;
        }),
    },
} satisfies Routes;
```
:::

#### `.finally(handler)`

Registers a cleanup callback that always runs, whether the request succeeds or fails.

:::code-group
```ts [Client]
routes.users.create(newUser).call()
    .then((user) => showSuccess(user))
    .catchUnknown((error) => showError(error))
    .finally(() => setLoading(false));
```

```ts [server.routes.ts]
type User = {id: string; name: string};

const routes = {
    users: {
        create: route((ctx, userData: Omit<User, 'id'>): User => {
            return {id: 'USER-123', ...userData};
        }),
    },
} satisfies Routes;
```
:::

## Hook Error Handling

Hooks support two different invocation methods, each returning a different type for error handling:

| Method | Returns | Use Case |
|--------|---------|----------|
| `.prefill()` | `TypedEvent` | Persistent handlers - hook data sent with every request |
| `.call()` | `TypedPromise` | Per-request handlers - same as routes |

### Using `.prefill()` with TypedEvent

When you use `.prefill()`, the hook data is stored and automatically sent with **every subsequent request**. Error handlers registered with `.onError()` persist and are called each time that error occurs across all requests.

This is ideal for authentication tokens, session data, or any hook data that should be included in all requests.

:::code-group
```ts [Client]
import {initClient} from '@mionkit/client';
import type {MyApi} from './server.routes';

const {hooks} = initClient<MyApi>({baseURL: 'http://localhost:3000'});

// Register persistent error handler for auth hook
hooks.auth('myToken-XYZ')
    .prefill()
    .onError('invalid-token', (error) => {
        // Called every time this error occurs (across ALL requests)
        redirectToLogin();
    })
    .onError('token-expired', (error) => {
        // Handle token expiration
        refreshToken();
    });
```

```ts [server.routes.ts]
import {RpcError} from '@mionkit/core';
import {Routes, headersHook, initMionRouter} from '@mionkit/router';

const routes = {
    auth: headersHook(
        ['authorization'],
        (ctx, token: string): void | RpcError<'invalid-token'> | RpcError<'token-expired'> => {
            if (!token) return new RpcError({type: 'invalid-token', statusCode: 401});
            if (isTokenExpired(token)) return new RpcError({type: 'token-expired', statusCode: 401});
        }
    ),
    // ... other routes
} satisfies Routes;

export const myApi = initMionRouter(routes);
export type MyApi = typeof myApi;
```
:::

### TypedEvent Methods

#### `.onError(errorType, handler)`

Registers a persistent error handler for a specific error type. This handler is called for ALL future requests that fail with this error from this hook.

:::code-group
```ts [Client]
hooks.credentials({token})
    .prefill()
    .onError('session-expired', (error) => {
        // Called every time session expires
        showSessionExpiredModal();
        redirectToLogin();
    });
```

```ts [server.routes.ts]
const routes = {
    credentials: hook(
        (ctx, creds: {token: string}): void | RpcError<'session-expired'> => {
            if (isSessionExpired(creds.token)) {
                return new RpcError({type: 'session-expired', statusCode: 401});
            }
        }
    ),
} satisfies Routes;
```
:::

#### `.offError(errorType)`

Removes a previously registered error handler.

:::code-group
```ts [Client]
// Remove the session-expired handler
hooks.credentials({token}).offError('session-expired');
```

```ts [server.routes.ts]
// Same server hook as above - offError is client-side only
const routes = {
    credentials: hook(
        (ctx, creds: {token: string}): void | RpcError<'session-expired'> => {
            if (isSessionExpired(creds.token)) {
                return new RpcError({type: 'session-expired', statusCode: 401});
            }
        }
    ),
} satisfies Routes;
```
:::

#### `.onSuccess(handler)`

Registers a persistent success handler called when the hook succeeds.

:::code-group
```ts [Client]
hooks.auth('myToken')
    .prefill()
    .onSuccess((result) => {
        console.log('Authentication successful');
    });
```

```ts [server.routes.ts]
const routes = {
    auth: headersHook(
        ['authorization'],
        (ctx, token: string): void => {
            // Validate token - no return value means success
            validateToken(token);
        }
    ),
} satisfies Routes;
```
:::

### Using `.hooks()` with Per-Request Hook Data

Instead of using `.prefill()`, you can pass hooks directly to a route call using the `.hooks()` method. This sends the hook data with that specific request only, without persisting it for future requests.

When using this pattern, you should:
1. Create the hook subrequest and store it in a variable
2. Register error handlers on the hook subrequest using `.onError()`
3. Pass the hook subrequest to the route's `.hooks()` method
4. Handle route-specific errors on the route's `.call()` TypedPromise

:::code-group
```ts [Client]
import {initClient} from '@mionkit/client';
import type {MyApi} from './server.routes';

const {routes, hooks} = initClient<MyApi>({baseURL: 'http://localhost:3000'});

// 1. Create and store the hook subrequest
const authHook = hooks.auth('temporary-token');

// 2. Register error handlers on the hook itself
authHook
    .onError('invalid-token', (error) => {
        // Handle hook error
        console.log('Token invalid for this request');
        redirectToLogin();
    })
    .onError('token-expired', (error) => {
        // Handle another hook error
        console.log('Token expired');
    });

// 3. Pass hook to route and handle route-specific errors
const result = await routes.users
    .getById('123')
    .hooks(authHook)  // Hook included for this request only
    .call()
    .then((user) => {
        console.log('User found:', user.name);
        return user;
    })
    .catchError('user-not-found', (error) => {
        // Handle route error only
        console.log('User not found:', error.errorData?.userId);
    })
    .catchUnknown((error) => {
        // Catches unhandled errors from route (hook errors handled above)
        console.log('Unexpected error:', error.publicMessage);
    });
```

```ts [server.routes.ts]
import {RpcError} from '@mionkit/core';
import {Routes, headersHook, route, initMionRouter} from '@mionkit/router';

type User = {id: string; name: string};

const routes = {
    auth: headersHook(
        ['authorization'],
        (ctx, token: string): void | RpcError<'invalid-token'> | RpcError<'token-expired'> => {
            if (!token) return new RpcError({type: 'invalid-token', statusCode: 401});
            if (isTokenExpired(token)) return new RpcError({type: 'token-expired', statusCode: 401});
        }
    ),
    users: {
        getById: route((ctx, id: string): User | RpcError<'user-not-found', {userId: string}> => {
            const user = findUser(id);
            if (!user) return new RpcError({type: 'user-not-found', errorData: {userId: id}});
            return user;
        }),
    },
} satisfies Routes;

export const myApi = initMionRouter(routes);
export type MyApi = typeof myApi;
```
:::

This is useful for:
- **Temporary tokens**: Password reset links, one-time access tokens
- **Different credentials**: Using a different auth token for a specific operation
- **Optional hooks**: Including a hook only when needed for certain requests

## Hooks: `.prefill()` vs `.call()` Comparison

Understanding when to use each method is crucial for proper error handling:

| Aspect | `.prefill()` → TypedEvent | `.call()` → TypedPromise |
|--------|---------------------------|--------------------------|
| **Data Persistence** | Stored and sent with ALL future requests | Sent with this request only |
| **Handler Scope** | Global - persist across all requests | Per-request - only for this call |
| **Registration** | `.onError()` - registered once | `.catchError()` - registered per call |
| **Use Case** | Auth tokens, session data | One-off operations, temporary tokens |
| **Examples** | Login token stored for session | Password reset, temporary access |

## Error Suppression Logic

When a hook error is handled by `.onError()`, it **suppresses** the route's `.catchUnknown()` handler. This prevents duplicate error messages.

### How Suppression Works

:::code-group
```ts [Client]
import {initClient} from '@mionkit/client';
import type {MyApi} from './server.routes';

const {routes, hooks} = initClient<MyApi>({baseURL: 'http://localhost:3000'});

// 1. Register persistent handler for auth errors
hooks.credentials({token})
    .prefill()
    .onError('invalid-token', (error) => {
        // This handles the error globally
        showToast('Session expired. Please login again.');
        redirectToLogin();
    });

// 2. Make a route call
routes.orders.create(orderData)
    .call()
    .then((order) => showSuccess(order))
    .catchError('cart-empty', (e) => showCartError())
    .catchUnknown((error) => {
        // This is NOT called for 'invalid-token' errors
        // because onError() already handled it
        showGenericError(error);
    });
```

```ts [server.routes.ts]
import {RpcError} from '@mionkit/core';
import {Routes, hook, route, initMionRouter} from '@mionkit/router';

type Order = {id: string; total: number};

const routes = {
    credentials: hook(
        (ctx, creds: {token: string}): void | RpcError<'invalid-token'> => {
            if (!isValidToken(creds.token)) {
                return new RpcError({type: 'invalid-token', statusCode: 401});
            }
        }
    ),
    orders: {
        create: route((ctx, data: OrderData): Order | RpcError<'cart-empty'> => {
            if (isCartEmpty(data)) return new RpcError({type: 'cart-empty'});
            return {id: 'ORDER-123', total: 99.99};
        }),
    },
} satisfies Routes;

export const myApi = initMionRouter(routes);
export type MyApi = typeof myApi;
```
:::

### Suppression Flow

1. Request is made with route and hooks
2. If a hook fails, the Client checks for registered `.onError()` handlers
3. If `.onError()` handles the error, it marks it as "handled"
4. The route's `.catchUnknown()` skips handled errors
5. Only truly unhandled errors reach `.catchUnknown()`

### Benefits

- **No duplicate error messages**: Auth errors show one notification, not two
- **Separation of concerns**: Handle auth globally, business logic per-request
- **Predictable behavior**: Each error is handled exactly once

## Complete Example

:::code-group
```ts [Client]
import {initClient} from '@mionkit/client';
import type {MyApi} from './server.routes';

const {routes, hooks} = initClient<MyApi>({baseURL: 'http://localhost:3000'});

// ===== SETUP: Register global hook error handlers (once) =====
hooks.auth('myToken-XYZ')
    .prefill()
    .onError('invalid-token', () => {
        showToast('Please login again');
        redirectToLogin();
    })
    .onError('token-expired', async () => {
        const newToken = await refreshToken();
        hooks.auth(newToken).prefill();
    });

// ===== USAGE: Handle route-specific errors per request =====
async function createOrder(cartData: CartData) {
    setLoading(true);

    await routes.checkout
        .createOrder(cartData)
        .call()
        .then((order) => {
            showSuccess(`Order ${order.id} created!`);
            navigateToOrder(order.id);
        })
        .catchError('cart-empty', () => {
            showWarning('Your cart is empty');
        })
        .catchError('payment-declined', (error) => {
            showError(`Payment failed: ${error.errorData?.reason}`);
        })
        .catchUnknown((error) => {
            // Only catches errors NOT handled by hooks.onError()
            showError('Something went wrong');
            logError(error);
        })
        .finally(() => {
            setLoading(false);
        });
}
```

```ts [server.routes.ts]
import {RpcError} from '@mionkit/core';
import {Routes, headersHook, route, initMionRouter} from '@mionkit/router';

type CartData = {items: {productId: string; quantity: number}[]};
type Order = {id: string; total: number};

const routes = {
    auth: headersHook(
        ['authorization'],
        (ctx, token: string): void | RpcError<'invalid-token'> | RpcError<'token-expired'> => {
            if (!token) return new RpcError({type: 'invalid-token', statusCode: 401});
            if (isTokenExpired(token)) return new RpcError({type: 'token-expired', statusCode: 401});
        }
    ),
    checkout: {
        createOrder: route((ctx, cartData: CartData): Order | RpcError<'cart-empty'> | RpcError<'payment-declined', {reason: string}> => {
            if (cartData.items.length === 0) {
                return new RpcError({type: 'cart-empty'});
            }
            const paymentResult = processPayment(cartData);
            if (!paymentResult.success) {
                return new RpcError({type: 'payment-declined', errorData: {reason: paymentResult.reason}});
            }
            return {id: 'ORDER-123', total: calculateTotal(cartData)};
        }),
    },
} satisfies Routes;

export const myApi = initMionRouter(routes);
export type MyApi = typeof myApi;
```
:::

## Type Reference

### TypedPromise

```ts
class TypedPromise<S, E extends RpcError<string, any> = never> implements PromiseLike<S> {
    /** Register handler for a specific error type */
    catchError<T extends E['type']>(
        errorType: T,
        handler: (error: Extract<E, {type: T}>) => void
    ): TypedPromise<S, E>;

    /** Register handler for any unhandled error */
    catchUnknown(handler: (error: RpcError<string, any>) => void): TypedPromise<S, E>;

    /** Register handler for all unhandled errors as a record */
    catch(handler: (errors: Record<string, RpcError<string, any>>) => void): TypedPromise<S, E>;

    /** Register cleanup callback that always runs */
    finally(onFinally: () => void): TypedPromise<S, E>;
}
```

### TypedEvent

```ts
class TypedEvent<S = void, E extends RpcError<string, any> = never> {
    /** Register persistent success handler */
    onSuccess(handler: (result: S) => void): TypedEvent<S, E>;

    /** Register persistent error handler for specific error type */
    onError<T extends E['type']>(
        errorType: T,
        handler: (error: Extract<E, {type: T}>) => void
    ): TypedEvent<S, E>;

    /** Remove a registered error handler */
    offError<T extends E['type']>(errorType: T): TypedEvent<S, E>;
}
```

### RpcError

```ts
class RpcError<T extends string = string, ErrData = unknown> extends Error {
    /** Error type identifier for typed handling */
    type: T;
    /** HTTP status code */
    statusCode: number;
    /** Public message sent to client */
    publicMessage: string;
    /** Optional typed error data */
    errorData?: ErrData;
}
