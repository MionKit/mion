The mion client uses a **Result pattern** for error handling. All API calls return a 4-tuple `[routeResult, routeError, hooksResults, hooksErrors]` - they **never throw**. This provides type-safe error handling with full TypeScript support.

## The Result Pattern

Every `call()` and `callWithHooks()` returns a 4-tuple:

```ts
type Result<RouteSuccess, RouteError, HooksResults, HooksErrors> = [
    RouteSuccess | undefined,
    RouteError | undefined,
    HooksResults | undefined,
    HooksErrors | undefined
];
```

This means:
- Position 0: Route result (if successful)
- Position 1: Route error (if failed)
- Position 2: Hook results (keyed by hook name)
- Position 3: Hook errors (keyed by hook name)

### Basic Usage

```ts
import {initClient} from '@mionkit/client';
import type {MyApi} from './server.routes';

const {routes, hooks} = initClient<MyApi>({baseURL: 'http://localhost:3000'});

// call() returns 4-tuple - never throws
const [user, error] = await routes.users.getById('123').call();

if (error) {
    // Handle error - TypeScript knows the error type
    console.log('Error:', error.publicMessage);
} else {
    // Handle success
    console.log('User:', user?.name);
}
```

## Handling Typed Errors

When routes define specific error types, TypeScript provides full autocompletion and type checking:

:::code-group
```ts [Client]
import {initClient} from '@mionkit/client';
import type {MyApi} from './server.routes';

const {routes} = initClient<MyApi>({baseURL: 'http://localhost:3000'});

const [user, error] = await routes.users.getById('123').call();

if (error) {
    // TypeScript knows error.type can be 'user-not-found'
    if (error.type === 'user-not-found') {
        // error.errorData is typed as {userId: string}
        console.log('User not found:', error.errorData?.userId);
    } else {
        console.log('Other error:', error.publicMessage);
    }
    return;
}

// TypeScript knows user is User here
console.log('Found user:', user.name);
```

```ts [server.routes.ts]
import {RpcError} from '@mionkit/core';
import {Routes, route, initMionRouter} from '@mionkit/router';

type User = {id: string; name: string};

const routes = {
    users: {
        getById: route((ctx, id: string): User | RpcError<'user-not-found', {userId: string}> => {
            const user = findUser(id);
            if (!user) return new RpcError({type: 'user-not-found', errorData: {userId: id}});
            return user;
        }),
    },
} satisfies Routes;

export const myApi = initMionRouter(routes);
export type MyApi = typeof myApi;
```
:::

## Using callWithHooks

When you need to pass hook data with a route call (like authentication), use `callWithHooks()`. This method returns a 4-tuple that includes both route and hook results.

### CallWithHooksResult Structure

```ts
type CallWithHooksResult<RouteSuccess, RouteError, Hooks> = [
    RouteSuccess | undefined,
    RouteError | undefined,
    { [K in keyof Hooks]?: HookSuccess<Hooks[K]> } | undefined,
    { [K in keyof Hooks]?: HookError<Hooks[K]> } | undefined,
];
```

### Basic callWithHooks Usage

:::code-group
```ts [Client]
import {initClient} from '@mionkit/client';
import type {MyApi} from './server.routes';

const {routes, hooks} = initClient<MyApi>({baseURL: 'http://localhost:3000'});

// Call route with auth hook - returns 4-tuple
const [user, routeError, hookResults, hookErrors] = await routes.users.getById('123').callWithHooks({
    auth: hooks.auth('myToken-XYZ'),
});

// Check for hook errors
if (hookErrors?.auth) {
    if (hookErrors.auth.type === 'invalid-token') {
        console.log('Invalid token, redirecting to login');
        redirectToLogin();
        return;
    }
    if (hookErrors.auth.type === 'token-expired') {
        console.log('Token expired, refreshing...');
        await refreshToken();
        return;
    }
}

// Check for route errors
if (routeError) {
    if (routeError.type === 'user-not-found') {
        console.log('User not found:', routeError.errorData?.userId);
        return;
    }
}

// Handle success
if (user) {
    console.log('User:', user.name);
}
```

```ts [server.routes.ts]
import {RpcError} from '@mionkit/core';
import {Routes, headersHook, route, initMionRouter} from '@mionkit/router';

type User = {id: string; name: string};

const routes = {
    auth: headersHook(
        ['authorization'],
        (ctx, token: string): void | RpcError<'invalid-token'> | RpcError<'token-expired'> => {
            if (!token) return new RpcError({type: 'invalid-token', statusCode: 401});
            if (isTokenExpired(token)) return new RpcError({type: 'token-expired', statusCode: 401});
        }
    ),
    users: {
        getById: route((ctx, id: string): User | RpcError<'user-not-found', {userId: string}> => {
            const user = findUser(id);
            if (!user) return new RpcError({type: 'user-not-found', errorData: {userId: id}});
            return user;
        }),
    },
} satisfies Routes;

export const myApi = initMionRouter(routes);
export type MyApi = typeof myApi;
```
:::

## Prefilling Hooks

For hooks that should be included in all requests (like authentication), use `prefill()`:

```ts
// Prefill auth hook - will be automatically included in all future requests
await hooks.auth('myToken-XYZ').prefill();

// Now call() works without explicitly passing auth
const [user, error] = await routes.users.getById('123').call();
```

### TypedEvent for Prefilled Hooks

When you prefill a hook, you can register persistent success/error handlers using `TypedEvent`:

```ts
const authHook = hooks.auth('myToken-XYZ');

// Register persistent success/error handlers BEFORE prefilling
authHook
    .onSuccess((sessionInfo) => {
        console.log('Authenticated as:', sessionInfo?.userId);
    })
    .onError('invalid-token', (error) => {
        // Called every time this error occurs across ALL requests
        redirectToLogin();
    })
    .onError('token-expired', (error) => {
        refreshToken();
    });

// Prefill the hook
await authHook.prefill();
```

## Prefill vs callWithHooks

| Aspect | `.prefill()` | `.callWithHooks()` |
|--------|--------------|-------------------|
| **Persistence** | Hook data stored for ALL future requests | Hook data sent with this request only |
| **Error Handlers** | Persistent via `.onError()` | Check `hookErrors` in 4-tuple |
| **Use Case** | Auth tokens, session data | Metadata related to the request |

## Complete Example

:::code-group
```ts [Client]
import {initClient} from '@mionkit/client';
import type {MyApi} from './server.routes';

const {routes, hooks} = initClient<MyApi>({baseURL: 'http://localhost:3000'});

// ===== Option 1: Prefill for persistent auth =====
const authHook = hooks.auth('myToken-XYZ');
authHook
    .onError('invalid-token', () => {
        showToast('Please login again');
        redirectToLogin();
    })
    .onError('token-expired', async () => {
        const newToken = await refreshToken();
        hooks.auth(newToken).prefill();
    });
await authHook.prefill();

// Now just use call() - auth is automatic
async function getUser(id: string) {
    const [user, error] = await routes.users.getById(id).call();

    if (error) {
        if (error.type === 'user-not-found') {
            showWarning(`User ${error.errorData?.userId} not found`);
        }
        return null;
    }

    return user;
}

// ===== Option 2: callWithHooks for per-request auth =====
async function createOrderWithTempToken(cartData: CartData, tempToken: string) {
    // Returns 4-tuple: [routeResult, routeError, hooksResults, hooksErrors]
    const [order, routeError, , hookErrors] = await routes.checkout.createOrder(cartData).callWithHooks({
        auth: hooks.auth(tempToken),
    });

    if (hookErrors?.auth) {
        showError('Authentication failed');
        return null;
    }

    if (routeError) {
        if (routeError.type === 'cart-empty') {
            showWarning('Your cart is empty');
        } else if (routeError.type === 'payment-declined') {
            showError(`Payment failed: ${routeError.errorData?.reason}`);
        }
        return null;
    }

    showSuccess(`Order ${order?.id} created!`);
    return order;
}
```

```ts [server.routes.ts]
import {RpcError} from '@mionkit/core';
import {Routes, headersHook, route, initMionRouter} from '@mionkit/router';

type User = {id: string; name: string};
type CartData = {items: {productId: string; quantity: number}[]};
type Order = {id: string; total: number};

const routes = {
    auth: headersHook(
        ['authorization'],
        (ctx, token: string): void | RpcError<'invalid-token'> | RpcError<'token-expired'> => {
            if (!token) return new RpcError({type: 'invalid-token', statusCode: 401});
            if (isTokenExpired(token)) return new RpcError({type: 'token-expired', statusCode: 401});
        }
    ),
    users: {
        getById: route((ctx, id: string): User | RpcError<'user-not-found', {userId: string}> => {
            const user = findUser(id);
            if (!user) return new RpcError({type: 'user-not-found', errorData: {userId: id}});
            return user;
        }),
    },
    checkout: {
        createOrder: route((ctx, cartData: CartData): Order | RpcError<'cart-empty'> | RpcError<'payment-declined', {reason: string}> => {
            if (cartData.items.length === 0) {
                return new RpcError({type: 'cart-empty'});
            }
            const paymentResult = processPayment(cartData);
            if (!paymentResult.success) {
                return new RpcError({type: 'payment-declined', errorData: {reason: paymentResult.reason}});
            }
            return {id: 'ORDER-123', total: calculateTotal(cartData)};
        }),
    },
} satisfies Routes;

export const myApi = initMionRouter(routes);
export type MyApi = typeof myApi;
```
:::

## Type Reference

### Result

```ts
type Result<RouteSuccess, RouteError, HooksResults, HooksErrors> = [
    RouteSuccess | undefined,
    RouteError | undefined,
    HooksResults | undefined,
    HooksErrors | undefined
];
```

### CallWithHooksResult

```ts
type CallWithHooksResult<RouteSuccess, RouteError, Hooks> = [
    RouteSuccess | undefined,
    RouteError | undefined,
    { [K in keyof Hooks]?: HookSuccess<Hooks[K]> } | undefined,
    { [K in keyof Hooks]?: HookError<Hooks[K]> } | undefined,
];
```

### TypedEvent

```ts
class TypedEvent<S = void, E extends RpcError<string, any> = never> {
    /** Register persistent success handler */
    onSuccess(handler: (result: S) => void): TypedEvent<S, E>;

    /** Register persistent error handler for specific error type */
    onError<T extends E['type']>(
        errorType: T,
        handler: (error: Extract<E, {type: T}>) => void
    ): TypedEvent<S, E>;

    /** Remove a registered error handler */
    offError<T extends E['type']>(errorType: T): TypedEvent<S, E>;

    /** Remove success handler */
    offSuccess(): TypedEvent<S, E>;
}
```

### RpcError

```ts
class RpcError<T extends string = string, ErrData = unknown> extends Error {
    /** Error type identifier for typed handling */
    type: T;
    /** HTTP status code */
    statusCode: number;
    /** Public message sent to client */
    publicMessage: string;
    /** Optional typed error data */
    errorData?: ErrData;
}
```
